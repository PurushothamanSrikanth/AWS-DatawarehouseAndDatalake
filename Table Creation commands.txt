#for Data Warehouse

CREATE TABLE test.users (
id STRING,
Name STRING,
Address STRING,
PAN STRING,
Phone_Number STRING,
isStudent STRING,
Salary STRING)
PARTITIONED BY (Month STRING)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
LOCATION '';



#for Data Lake

CREATE EXTERNAL TABLE test.users (
id STRING,
Name STRING,
Address STRING,
PAN STRING,
Phone_Number INT(10),
isStudent BOOLEAN,
Salary INT)
PARTITIONED BY (Month STRING)
STORED AS PARQUET
LOCATION '';
